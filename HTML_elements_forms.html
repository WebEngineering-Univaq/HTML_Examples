Nuove Tipologie di input in HTML5

Input (con autocomplete):

<input type="text" name="i33" value="valore" autocomplete="on"/>

Input (con pattern):

<input type="text" name="i34" value="IDabc" pattern="ID[a-z]+"/>

Input (con placeholder):

<input type="text" name="i35" placeholder="Inserire qualcosa..."/>

Input (required):

<input type="text" value="qualcosa" name="i36" required="required"/>

Input (type=tel):

<input type="tel" name="i37" value="333444" />

Input (type=search):

<input type="search" name="i38" value="valore" />

Input (type=url):

<input type="url" name="i39" value="http://pippo.it/pluto" />

Input (type=email):

<input type="email" name="i40" value="gdp@pippo.net" />

Input (type=number, con min, max e step):

<input type="number" name="i41" value="4" min="2" max="8" step="2"/>

Input (type=range, con min, max e step):

<input type="range" name="i42" value="4" min="2" max="8" step="2" />

Input (type=color):

<input type="color" name="i43" value="#555555" />

Input (type=date, con min e max):

<input type="date" name="i44" min="2015-01-01" max="2020-12-31" />

Input (type=time):

<input type="time" name="i45"/>

Input (type=email, multiplo):

<input type="email" name="i46" value="gdp@pippo.com, gdp1@pluto.it" multiple="multiple" />

Input (type=file, multiplo):

<input type="file" name="i47" multiple="multiple" />
