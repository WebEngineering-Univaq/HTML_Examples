Nuove Tipologie di input

Input (con autocomplete):

<input type="text" name="i0" value="valore" autocomplete="on"/>

Input (con pattern):

<input type="text" name="i1" value="IDabc" pattern="ID[a-z]+"/>

Input (con placeholder):

<input type="text" name="i2" placeholder="Inserire qualcosa..."/>

Input (required):

<input type="text" value="qualcosa" name="i3" required="required"/>

Input (type=tel):

<input type="tel" name="i4" value="333444" />

Input (type=search):

<input type="search" name="i5" value="valore" />

Input (type=url):

<input type="url" name="i6" value="http://pippo.it/pluto" />

Input (type=email):

<input type="email" name="i7" value="gdp@pippo.net" />

Input (type=number, con min, max e step):

<input type="number" name="i8" value="4" min="2" max="8" step="2"/>

Input (type=range, con min, max e step):

<input type="range" name="i9" value="4" min="2" max="8" step="2" />

Input (type=color):

<input type="color" name="i10" value="#555555" />

Input (type=date, con min e max):

<input type="date" name="i11" min="2015-01-01" max="2020-12-31" />

Input (type=time):

<input type="time" name="i12"/>

Input (type=email, multiplo):

<input type="email" name="i13" value="gdp@pippo.com, gdp1@pluto.it" multiple="multiple" />

Input (type=file, multiplo):

<input type="file" name="i14" multiple="multiple" />

Input (type=submit con formcation e formnovalidate):

<input type="submit" name="i15" value="submit SENZA validazione" formaction="pippo.php" formnovalidate="formnovalidate" />

Input (type=submit):

<input type="submit" name="i16" value="submit CON validazione" />
